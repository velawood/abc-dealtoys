3. Create `src/components/ScrollReveal.svelte` -- a reusable animation wrapper using svelte-inview:

   ```svelte
   <script lang="ts">
     import { inview } from "svelte-inview";

     interface Props {
       threshold?: number;
       delay?: number;
       animation?: "fade-up" | "fade-in" | "fade-left" | "fade-right";
       children?: import("svelte").Snippet;
     }

     let {
       threshold = 0.2,
       delay = 0,
       animation = "fade-up",
       children,
     }: Props = $props();

     let isVisible = $state(false);

     function handleInview(event: CustomEvent<ObserverEventDetails>) {
       if (event.detail.inView) {
         isVisible = true;
       }
     }
   </script>

   <div
     use:inview={{ threshold }}
     oninview_enter={handleInview}
     class="scroll-reveal {animation}"
     class:visible={isVisible}
     style="transition-delay: {delay}ms"
   >
     {@render children?.()}
   </div>

   <style>
     .scroll-reveal {
       transition: opacity 0.7s ease-out, transform 0.7s ease-out;
     }

     .scroll-reveal.fade-up {
       opacity: 0;
       transform: translateY(2rem);
     }
     .scroll-reveal.fade-in {
       opacity: 0;
     }
     .scroll-reveal.fade-left {
       opacity: 0;
       transform: translateX(-2rem);
     }
     .scroll-reveal.fade-right {
       opacity: 0;
       transform: translateX(2rem);
     }

     .scroll-reveal.visible {
       opacity: 1;
       transform: translate(0, 0);
     }
   </style>
   ```

   IMPORTANT NOTES:
   - Use Svelte 5 syntax: `$props()`, `$state()`, `{@render children?.()}` for slot content.
   - The `use:inview` directive is from svelte-inview. Check the v4 API -- it may use `on:inview_enter` or `oninview_enter` depending on the Svelte 5 compatibility. If the package exports an `inview` action, it uses Svelte's `use:` directive. The callback event name may need adjustment -- check the package docs/types after install.
   - The component starts with opacity:0 and transform offset, then transitions to visible state. This prevents layout shift because the element occupies its full space from the start (only opacity and transform change, not dimensions).
   - The `threshold: 0.2` means 20% of the element must be visible before triggering.
   - `delay` prop allows staggering multiple items.
